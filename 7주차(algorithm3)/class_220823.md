### 1223.계산기2
```python
T = 10

icp = {"+": 1, "*": 2}

for tc in range(1, T + 1):
    n = int(input()) # 식의 길이

    exp = input() # 중위표현식으로 입력받은 문자
    postfix = ""  # 후위표현식으로 바꾼 결과

    stack = []
    for i in range(n):
        # 글자를 한자리씩 보면서 연산자와 피연산자로 구분
        if "0" <= exp[i] <= "9": # 피연산자면 그냥 이어 붙이기
            postfix += exp[i]
        else:
            # 연산자면 스택의 꼭대기를 확인해서
            # 자신보다 같거나 높은 우선순위를 가진 연산자가 있으면 꺼낸다.
            if stack and icp[stack[-1]] >= icp[exp[i]]:
                postfix += stack.pop()
            stack.append(exp[i])

    # 스택에 남아있는 연산자는 뒤에 붙이기
    while stack:
        postfix += stack.pop()

    # 후위표기식 계산
    stack = []
    result = 0
    # print(postfix)
    k = len(postfix)
    for i in range(k):
        # 피연산자가 나오면 그냥 넘어가고
        # 스택에다가 담는다
        if "0" <= postfix[i] <= "9":
            stack.append(postfix[i])
        else:
            # 연산자가 나오면 계산(앞에 두 피연산자를 선택)
            # 스택에서 두개의 피연산자를 꺼낸다.
            right = int(stack.pop())
            left = int(stack.pop())

            if postfix[i] == "+":
                result = right + left
            else:
                result = right * left

            stack.append(result)

    print(f"#{tc} {result}")
```

### 토너먼트 카드게임
```python
# 카드게임 분할정복 함수
# i, j : i번째 학생부터 j번째 학생까지
def tournament(i, j):
    if i == j:
        # i랑 j가 같으면 둘로 쪼개는게 불가능하다
        return i
    # 왼쪽과 오른쪽을 나누는 일은 계속 반복적으로 하니까 재귀함수로 만든다
    else:
        left = tournament(i, (i + j) // 2)  # 왼쪽 쪼개기
        right = tournament((i + j) // 2 + 1, j)  # 오른쪽 쪼개기
        return winner(left, right)  # left와 right 중 승자를 구해서 리턴

# 왼쪽과 오른쪽 중에 승자를 정한다
# 승자의 인덱스를 리턴
def winner(left, right):
    if numbers[left] == 1:
        if numbers[right] == 2:
            return right
        else:
            return left
    elif numbers[left] == 2:
        if numbers[right] == 3:
            return right
        else:
            return left
    elif numbers[left] == 3:
        if numbers[right] == 1:
            return right
        else:
            return left

T = int(input())
for tc in range(1, T+1):
    n = int(input())
    numbers = list(map(int,input().split()))
    print(f'#{tc} {tournament(0,n-1)+1}')
```
